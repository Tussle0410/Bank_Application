create table user(
ID varchar(20) not null,
PW varchar(20) not null,
Name varchar(20) not null,
Gender varchar(5) not null,
Email varchar(25) not null,
Birth date not null,
primary key(ID,Email)
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

create table accountAddress(
userID varchar(20) not null,
address varchar(20) not null,
kinds varchar(20) default 'deposit',
production_no int default 0,
money bigint not null default 0,
remittanceLimit int defalut 500000,
currentremittance int default 0,
addressName varchar(20) default 'deposit',
mainAddressCheck Tinyint(1) not null default 0,
primary key(address),
foreign key(userID) references user(ID),
foreign key(kinds) references production(kinds)
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

create table transactionHistory(
sendAddress varchar(20) not null,
receiveAddress varchar(20) not null,
money bigint not null,
kinds varchar(20) default 'deposit',
transactionDate date not null,
foreign key(sendAddress,recieveAddress) references accountaddress(address),
foreign key(kinds) references production(kinds)
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

create table production(
productionName varchar(20) not null,
kinds varchar(20) not null,
productionNo int not null,
interestRate float(5,2) not null,
primary key(kinds) 
)ENGINE=InnoDB DEFAULT CHARSET=utf8;

create table banner(
bannerRoute varchar(20) not null,
kinds varchar(20) not null,
proudctionNo int not null,
primary key(bannerRoute),
foreign key(kinds, proudctionNo) references production(kinds,productionNo)
)ENGINE=InnoDB DEFAULT CHARSET=utf8;
